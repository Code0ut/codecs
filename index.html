WAV Analyzer – Python Code

import numpy as np
import librosa
import matplotlib.pyplot as plt
from scipy.signal import butter, filtfilt, hilbert
from scipy.signal import find_peaks


def load_wav(file_path):
    signal, fs = librosa.load(file_path, sr=None, mono=True)
    signal = signal - np.mean(signal)
    signal = signal / np.max(np.abs(signal))
    return signal, fs

def bandpass_filter(signal, fs, lowcut=300, highcut=3000, order=4):
    nyq = 0.5 * fs
    low = lowcut / nyq
    high = highcut / nyq
    b, a = butter(order, [low, high], btype='band')
    return filtfilt(b, a, signal)

def envelope_detection(signal):
    analytic_signal = hilbert(signal)
    envelope = np.abs(analytic_signal)
    return envelope

def lowpass_filter(signal, fs, cutoff=200, order=4):
    nyq = 0.5 * fs
    norm_cutoff = cutoff / nyq
    b, a = butter(order, norm_cutoff, btype='low')
    return filtfilt(b, a, signal)

def demon_spectrum(envelope, fs):
    n = len(envelope)
    spectrum = np.abs(np.fft.rfft(envelope))
    freqs = np.fft.rfftfreq(n, d=1/fs)
    return freqs, spectrum

def plot_demon(freqs, spectrum, max_freq=200):
    idx = freqs <= max_freq
    plt.figure(figsize=(10, 4))
    plt.plot(freqs[idx], spectrum[idx])
    plt.xlabel("Frequency (Hz)")
    plt.ylabel("Magnitude")
    plt.title("DEMON Spectrum")
    plt.grid()
    plt.show()

def demon_analysis(file_path):
    signal, fs = load_wav(file_path)

    bp_signal = bandpass_filter(signal, fs)
    envelope = envelope_detection(bp_signal)
    envelope_lp = lowpass_filter(envelope, fs)

    freqs, spectrum = demon_spectrum(envelope_lp, fs)
    plot_demon(freqs, spectrum)

    return freqs, spectrum

def detect_shaft_frequencies(freqs, spectrum, max_freq=20, prominence=0.05):
    idx = freqs <= max_freq
    freqs_lf = freqs[idx]
    spec_lf = spectrum[idx] / np.max(spectrum[idx])

    peaks, props = find_peaks(spec_lf, prominence=prominence)
    shaft_freqs = freqs_lf[peaks]

    return shaft_freqs

def compute_srpm(shaft_freqs):
    return shaft_freqs * 60

def detect_blades(freqs, spectrum, shaft_freq, tol=0.2, max_harmonic=10):
    blade_candidates = []

    for k in range(2, max_harmonic + 1):
        target_freq = k * shaft_freq
        idx = np.where(np.abs(freqs - target_freq) <= tol)[0]
        if len(idx) > 0:
            blade_candidates.append(k)

    if blade_candidates:
        return blade_candidates[0]
    return None

def extract_mechanical_parameters(freqs, spectrum):
    shaft_freqs = detect_shaft_frequencies(freqs, spectrum)
    srpms = compute_srpm(shaft_freqs)

    results = []

    for i, f_shaft in enumerate(shaft_freqs):
        blades = detect_blades(freqs, spectrum, f_shaft)

        results.append({
            "shaft_id": i + 1,
            "shaft_frequency_hz": round(f_shaft, 2),
            "shaft_rpm": round(f_shaft * 60, 1),
            "number_of_blades": blades
        })

    return results

file_path = "humpback_whale.wav"
freqs, spectrum = demon_analysis(file_path)
results = extract_mechanical_parameters(freqs, spectrum)
for r in results:
    print(r)

    x,
    fs=fs,
    window=window,
    nperseg=nperseg,
    noverlap=noverlap,
    nfft=n_fft,
    detrend=False,
    scaling='density',   # gives PSD (V^2/Hz) style units

import numpy as np
from scipy import signal
import matplotlib.pyplot as plt

import numpy as np
from scipy import signal

def third_octave_centers(fmin=20, fmax=20000, ref_fc=1000):
    n_low = int(np.ceil(3 * np.log2(fmin / ref_fc)))
    n_high = int(np.floor(3 * np.log2(fmax / ref_fc)))
    n = np.arange(n_low, n_high + 1)
    return ref_fc * (2 ** (n / 3.0))

def design_third_octave_filters(fc, fs, order=4):
    filters = []
    nyq = fs / 2.0
    bw_factor = 2 ** (1/6)  # 1/6 octave from center to edge

    for f_center in fc:
        f_low = f_center / bw_factor
        f_high = f_center * bw_factor

        # Skip bands that go outside (0, Nyquist)
        if f_low <= 0 or f_high >= nyq:
            continue

        low = f_low / nyq
        high = f_high / nyq

        # Extra safety check
        if not (0 < low < high < 1):
            continue

        b, a = signal.butter(order, [low, high], btype='band')
        filters.append((b, a, f_center))

    return filters

fc = third_octave_centers(20, fs/2) 

filters = design_third_octave_filters(fc, fs)

# Example signal (replace with your sonar data)
t = np.arange(0, 10, 1/fs)
x = np.random.randn(len(t))  # white noise example

# Filter signal through each band
band_signals = []
band_rms = []
valid_fc = []
for b, a, f_center in filters:
    # Filter the signal
    y_band = signal.filtfilt(b, a, x)  # zero-phase filtering
    
    # Compute RMS power in this band
    rms = np.sqrt(np.mean(y_band**2))
    band_signals.append(y_band)
    band_rms.append(rms)
    valid_fc.append(f_center)
band_rms_db = 20 * np.log10(np.array(band_rms) + 1e-12)
valid_fc = np.array(valid_fc)
plt.figure(figsize=(10, 6))
plt.semilogx(valid_fc, band_rms_db, 'o-', linewidth=2)

plt.xlim([20, fs/2])
plt.ylim([-60, 0])
plt.show()

def third_octave_analysis(x, fs, fmin=20, fmax=None):
    """Complete 1/3-octave analysis: returns centers, levels (dB), and band signals."""
    if fmax is None:
        fmax = fs / 2
    fc = third_octave_centers(fmin, fmax)
    filters = design_third_octave_filters(fc, fs)
    
    band_rms = []
    for b, a, f_center in filters:
        y_band = signal.filtfilt(b, a, x)
        rms = np.sqrt(np.mean(y_band**2))
        band_rms.append(rms)
    
    return fc, 20 * np.log10(np.array(band_rms) + 1e-12)

# Usage
fc, levels_db = third_octave_analysis(x, fs)
