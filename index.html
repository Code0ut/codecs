Introduction

This page documents a Python-based WAV file analyzer with metadata extraction, signal analysis, spectral features, and visualization.

WAV Analyzer – Python Code

import os
import wave
import numpy as np
import librosa
import librosa.display
import matplotlib.pyplot as plt
from scipy.stats import entropy


class WavAnalyzer:

    def __init__(self, file_path):
        self.file_path = file_path
        self.y = None
        self.sr = None
        self.metadata = {}
        self.features = {}

    def load_audio(self):
        self.y, self.sr = librosa.load(self.file_path, sr=None)
        return self.y, self.sr

    def extract_metadata(self):
        with wave.open(self.file_path, 'rb') as wav:
            self.metadata["file_name"] = os.path.basename(self.file_path)
            self.metadata["file_size_bytes"] = os.path.getsize(self.file_path)
            self.metadata["channels"] = wav.getnchannels()
            self.metadata["sample_width_bytes"] = wav.getsampwidth()
            self.metadata["sample_rate"] = wav.getframerate()
            self.metadata["num_frames"] = wav.getnframes()
            self.metadata["duration_seconds"] = (
                self.metadata["num_frames"] / self.metadata["sample_rate"]
            )
        return self.metadata

    def extract_signal_features(self):
        y = self.y
        self.features["min_amplitude"] = float(np.min(y))
        self.features["max_amplitude"] = float(np.max(y))
        self.features["mean_amplitude"] = float(np.mean(y))
        self.features["rms_amplitude"] = float(np.sqrt(np.mean(y**2)))
        self.features["signal_energy"] = float(np.sum(y**2))

        hist, _ = np.histogram(y, bins=256, density=True)
        self.features["signal_entropy"] = float(entropy(hist + 1e-10))

        zcr = librosa.feature.zero_crossing_rate(y)
        self.features["zero_crossing_rate"] = float(np.mean(zcr))

        return self.features

    def extract_spectral_features(self):
        y, sr = self.y, self.sr

        spectral_centroid = librosa.feature.spectral_centroid(y=y, sr=sr)
        spectral_bandwidth = librosa.feature.spectral_bandwidth(y=y, sr=sr)
        spectral_rolloff = librosa.feature.spectral_rolloff(y=y, sr=sr)
        mfcc = librosa.feature.mfcc(y=y, sr=sr, n_mfcc=13)

        self.features["spectral_centroid_mean"] = float(np.mean(spectral_centroid))
        self.features["spectral_bandwidth_mean"] = float(np.mean(spectral_bandwidth))
        self.features["spectral_rolloff_mean"] = float(np.mean(spectral_rolloff))
        self.features["mfcc_mean"] = np.mean(mfcc, axis=1).tolist()

        return self.features

    def visualize(self):
        y, sr = self.y, self.sr

        plt.figure(figsize=(14, 10))

        plt.subplot(3, 2, 1)
        librosa.display.waveshow(y, sr=sr)
        plt.title("Waveform")

        plt.subplot(3, 2, 2)
        D = librosa.amplitude_to_db(np.abs(librosa.stft(y)), ref=np.max)
        librosa.display.specshow(D, sr=sr, x_axis="time", y_axis="hz")
        plt.colorbar()
        plt.title("Spectrogram")

        plt.subplot(3, 2, 3)
        mel = librosa.feature.melspectrogram(y=y, sr=sr)
        mel_db = librosa.power_to_db(mel, ref=np.max)
        librosa.display.specshow(mel_db, sr=sr, x_axis="time", y_axis="mel")
        plt.colorbar()
        plt.title("Mel Spectrogram")

        plt.subplot(3, 2, 4)
        mfcc = librosa.feature.mfcc(y=y, sr=sr, n_mfcc=13)
        librosa.display.specshow(mfcc, x_axis="time")
        plt.colorbar()
        plt.title("MFCC")

        plt.subplot(3, 2, 5)
        zcr = librosa.feature.zero_crossing_rate(y)
        plt.plot(zcr[0])
        plt.title("Zero Crossing Rate")

        plt.tight_layout()
        plt.show()

    def analyze(self):
        self.load_audio()
        self.extract_metadata()
        self.extract_signal_features()
        self.extract_spectral_features()
        return {**self.metadata, **self.features}


if __name__ == "__main__":
    analyzer = WavAnalyzer("example.wav")
    results = analyzer.analyze()
    analyzer.visualize()

    for key, value in results.items():
        print(f"{key}: {value}")
