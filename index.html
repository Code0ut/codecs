WAV Analyzer – Python Code

import os
import wave
import numpy as np
import librosa
import librosa.display
import matplotlib.pyplot as plt
from scipy.stats import entropy


class WavAnalyzer:

    def __init__(self, file_path):
        self.file_path = file_path
        self.y = None
        self.sr = None
        self.metadata = {}
        self.features = {}

    # ---------- AUDIO LOADING ----------
    def load_audio(self):
        self.y, self.sr = librosa.load(self.file_path, sr=None)

    # ---------- METADATA ----------
    def extract_metadata(self):
        with wave.open(self.file_path, 'rb') as wav:
            self.metadata["file_name"] = os.path.basename(self.file_path)
            self.metadata["file_size_bytes"] = os.path.getsize(self.file_path)
            self.metadata["channels"] = wav.getnchannels()
            self.metadata["sample_width_bytes"] = wav.getsampwidth()
            self.metadata["sample_rate"] = wav.getframerate()
            self.metadata["num_frames"] = wav.getnframes()
            self.metadata["duration_seconds"] = (
                self.metadata["num_frames"] / self.metadata["sample_rate"]
            )

    # ---------- SIGNAL FEATURES ----------
    def get_min_amplitude(self):
        self.features["min_amplitude"] = float(np.min(self.y))

    def get_max_amplitude(self):
        self.features["max_amplitude"] = float(np.max(self.y))

    def get_mean_amplitude(self):
        self.features["mean_amplitude"] = float(np.mean(self.y))

    def get_rms_amplitude(self):
        self.features["rms_amplitude"] = float(np.sqrt(np.mean(self.y ** 2)))

    def get_signal_energy(self):
        self.features["signal_energy"] = float(np.sum(self.y ** 2))

    def get_signal_entropy(self):
        hist, _ = np.histogram(self.y, bins=256, density=True)
        self.features["signal_entropy"] = float(entropy(hist + 1e-10))

    def get_zero_crossing_rate(self):
        zcr = librosa.feature.zero_crossing_rate(self.y)
        self.features["zero_crossing_rate"] = float(np.mean(zcr))

    # ---------- SPECTRAL FEATURES ----------
    def get_spectral_centroid(self):
        centroid = librosa.feature.spectral_centroid(y=self.y, sr=self.sr)
        self.features["spectral_centroid_mean"] = float(np.mean(centroid))

    def get_spectral_bandwidth(self):
        bandwidth = librosa.feature.spectral_bandwidth(y=self.y, sr=self.sr)
        self.features["spectral_bandwidth_mean"] = float(np.mean(bandwidth))

    def get_spectral_rolloff(self):
        rolloff = librosa.feature.spectral_rolloff(y=self.y, sr=self.sr)
        self.features["spectral_rolloff_mean"] = float(np.mean(rolloff))

    def get_mfcc(self):
        mfcc = librosa.feature.mfcc(y=self.y, sr=self.sr, n_mfcc=13)
        self.features["mfcc_mean"] = np.mean(mfcc, axis=1).tolist()

    # ---------- FEATURE PIPELINE ----------
    def extract_signal_features(self):
        self.get_min_amplitude()
        self.get_max_amplitude()
        self.get_mean_amplitude()
        self.get_rms_amplitude()
        self.get_signal_energy()
        self.get_signal_entropy()
        self.get_zero_crossing_rate()

    def extract_spectral_features(self):
        self.get_spectral_centroid()
        self.get_spectral_bandwidth()
        self.get_spectral_rolloff()
        self.get_mfcc()


    # ---------- WAVEFORM ----------
    def plot_waveform(self, output_dir):
        plt.figure(figsize=(8, 4))
        librosa.display.waveshow(self.y, sr=self.sr)
        plt.title("Waveform")
        plt.tight_layout()

        path = os.path.join(output_dir, "waveform.png")
        plt.savefig(path)
        plt.close()
        return path

    # ---------- SPECTROGRAM ----------
    def plot_spectrogram(self, output_dir):
        plt.figure(figsize=(8, 4))
        D = librosa.amplitude_to_db(
            np.abs(librosa.stft(self.y)), ref=np.max
        )
        librosa.display.specshow(
            D, sr=self.sr, x_axis="time", y_axis="hz"
        )
        plt.colorbar(format="%+2.0f dB")
        plt.title("Spectrogram")
        plt.tight_layout()

        path = os.path.join(output_dir, "spectrogram.png")
        plt.savefig(path)
        plt.close()
        return path

    # ---------- MEL SPECTROGRAM ----------
    def plot_mel_spectrogram(self, output_dir):
        plt.figure(figsize=(8, 4))
        mel = librosa.feature.melspectrogram(y=self.y, sr=self.sr)
        mel_db = librosa.power_to_db(mel, ref=np.max)
        librosa.display.specshow(
            mel_db, sr=self.sr, x_axis="time", y_axis="mel"
        )
        plt.colorbar(format="%+2.0f dB")
        plt.title("Mel Spectrogram")
        plt.tight_layout()

        path = os.path.join(output_dir, "mel_spectrogram.png")
        plt.savefig(path)
        plt.close()
        return path

    # ---------- MFCC ----------
    def plot_mfcc(self, output_dir):
        plt.figure(figsize=(8, 4))
        mfcc = librosa.feature.mfcc(y=self.y, sr=self.sr, n_mfcc=13)
        librosa.display.specshow(mfcc, x_axis="time")
        plt.colorbar()
        plt.title("MFCC")
        plt.tight_layout()

        path = os.path.join(output_dir, "mfcc.png")
        plt.savefig(path)
        plt.close()
        return path

    # ---------- ZERO CROSSING RATE ----------
    def plot_zero_crossing_rate(self, output_dir):
        plt.figure(figsize=(8, 4))
        zcr = librosa.feature.zero_crossing_rate(self.y)
        plt.plot(zcr[0])
        plt.title("Zero Crossing Rate")
        plt.xlabel("Frames")
        plt.ylabel("Rate")
        plt.tight_layout()

        path = os.path.join(output_dir, "zcr.png")
        plt.savefig(path)
        plt.close()
        return path


    # ---------- MAIN ANALYSIS ----------
    def analyze(self):
        self.load_audio()
        self.extract_metadata()
        self.extract_signal_features()
        self.extract_spectral_features()
        return {**self.metadata, **self.features}

CSS Code Example

from reportlab.lib.pagesizes import A4
from reportlab.platypus import (
    SimpleDocTemplate,
    Paragraph,
    Spacer,
    Table,
    TableStyle,
    Image,
    PageBreak
)
from reportlab.lib.styles import getSampleStyleSheet
from reportlab.lib import colors
from reportlab.lib.units import inch


class PDFWriter:

    def __init__(self, output_path):
        self.output_path = output_path
        self.document = SimpleDocTemplate(
            output_path,
            pagesize=A4,
            rightMargin=40,
            leftMargin=40,
            topMargin=40,
            bottomMargin=40
        )
        self.elements = []
        self.styles = getSampleStyleSheet()

    # ---------- TITLE ----------
    def add_title(self, title):
        self.elements.append(
            Paragraph(f"{title}", self.styles["Title"])
        )
        self.elements.append(Spacer(1, 20))

    # ---------- METADATA ----------
    def add_metadata(self, metadata):
        self.elements.append(
            Paragraph("Audio File Metadata", self.styles["Heading2"])
        )
        self.elements.append(Spacer(1, 10))

        table_data = [["Parameter", "Value"]]
        for key, value in metadata.items():
            table_data.append([key, str(value)])

        table = Table(table_data, colWidths=[200, 250])
        table.setStyle(TableStyle([
            ("GRID", (0, 0), (-1, -1), 1, colors.black),
            ("BACKGROUND", (0, 0), (-1, 0), colors.lightgrey),
            ("ALIGN", (1, 1), (-1, -1), "LEFT"),
        ]))

        self.elements.append(table)
        self.elements.append(Spacer(1, 20))

    # ---------- FEATURES ----------
    def add_features(self, features):
        self.elements.append(
            Paragraph("Extracted Audio Features", self.styles["Heading2"])
        )
        self.elements.append(Spacer(1, 10))

        table_data = [["Feature", "Value"]]
        for key, value in features.items():
            table_data.append([key, str(value)])

        table = Table(table_data, colWidths=[200, 250])
        table.setStyle(TableStyle([
            ("GRID", (0, 0), (-1, -1), 1, colors.black),
            ("BACKGROUND", (0, 0), (-1, 0), colors.lightgrey),
        ]))

        self.elements.append(table)
        self.elements.append(PageBreak())

    # ---------- PLOTS ----------
    def add_plot(self, image_path, title):
        self.elements.append(
            Paragraph(f"{title}", self.styles["Heading2"])
        )
        self.elements.append(Spacer(1, 10))

        img = Image(image_path, width=5.5 * inch, height=3 * inch)
        self.elements.append(img)
        self.elements.append(Spacer(1, 20))

    # ---------- SAVE ----------
    def save(self):
        self.document.build(self.elements)
